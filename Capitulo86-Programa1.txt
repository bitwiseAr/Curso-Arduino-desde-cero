int valorBoton = 0;		// variable para almacenar valor leído de entrada A0

void setup()
{
  Serial.begin(9600);		// inicialización de monitor serie a 9600 baudios
}

void loop()
{
  valorBoton = analogRead(A0);	// lectura de entrada A0 y asignación a variable
  Serial.println(valorBoton);	// muestra por monitor serie el valor
  
  delay(500);			// demora de 500 milisegundos
}
